## ReAct 到底是什么
`ReAct` = `Reason`（推理） + `Act`（行动）

核心不是“一次回答”，而是**循环决策**：
1. 先判断当前信息是否足够
2. 不够就调用工具（搜索、查库、执行代码）
3. 读取工具结果
4. 再判断是否还缺信息
5. 直到能给出最终答案

```text
Question
  -> Thought(当前判断)
  -> Action(调用哪个工具 + 参数)
  -> Observation(工具返回结果)
  -> Thought(是否继续工具调用)
  -> ... (循环)
  -> Final Answer
```



## 代码示例

```python

"""
from __future__ import annotations
import json
import os
from datetime import datetime
from typing import Any
from openai import OpenAI

API_KEY = "sk-xxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxx"
BASE_URL = "https://dashscope.aliyuncs.com/compatible-mode/v1"

def get_current_time() -> str:
    return datetime.now().strftime("%Y-%m-%d %H:%M:%S")

def serper_search(query: str) -> str:
    # 先 mock，避免依赖外部搜索 key
    return f"[mock_search] 已检索关键词: {query}"


TOOLS = [
    {
        "type": "function",
        "function": {
            "name": "get_current_time",
            "description": "获取当前时间",
            "parameters": {"type": "object", "properties": {}, "required": []},
        },
    },
    {
        "type": "function",
        "function": {
            "name": "serper_search",
            "description": "互联网搜索",
            "parameters": {
                "type": "object",
                "properties": {"query": {"type": "string"}},
                "required": ["query"],
            },
        },
    },
]


def summarize_observation(raw: str, max_len: int = 120) -> str:
    """把工具输出压缩成适合前端显示的简短观察摘要。"""
    text = raw.replace("\n", " ").strip()
    if len(text) <= max_len:
        return text
    return text[:max_len] + "..."


def parse_json_safely(text: str) -> dict[str, Any] | None:
    """
    兼容两种返回：
    1) 纯 JSON
    2) ```json ... ``` 代码块
    """
    text = text.strip()
    if text.startswith("```"):
        text = text.removeprefix("```json").removeprefix("```").removesuffix("```").strip()
    try:
        return json.loads(text)
    except Exception:
        return None


def call_tool(tool_name: str, tool_args: dict[str, Any]) -> str:
    """统一的工具执行入口，便于后续扩展更多工具。"""
    if tool_name == "get_current_time":
        return get_current_time()
    if tool_name == "serper_search":
        return serper_search(str(tool_args.get("query", "")))
    return f"工具不存在: {tool_name}"


def run_react_with_visible_trace(user_input: str, model: str = "qwen-max", max_steps: int = 8) -> str:
    """
    核心做法：
    - 每一步先让模型输出“可展示摘要 + 是否调用工具 + 工具参数”（JSON）
    - 如果要调用工具，程序执行后把 observation 回填
    - 直到模型决定 final_answer
    """
    api_key = API_KEY if API_KEY.strip() else os.getenv("DASHSCOPE_API_KEY", "")
    if not api_key or "请替换" in api_key:
        raise RuntimeError("请先把文件顶部 API_KEY 改成你的真实 key。")
    client = OpenAI(
        api_key=api_key,
        base_url=BASE_URL,
    )

    # state_messages 存“事实状态”，每轮都回喂给模型，形成连续决策。
    state_messages: list[dict[str, str]] = [
        {"role": "user", "content": user_input},
    ]

    allowed_tool_names = {item["function"]["name"] for item in TOOLS}

    for step in range(1, max_steps + 1):
        planner_prompt = (
            "你是 ReAct Agent 的规划器。"
            "你每一步只做一个决策：要么调一个工具，要么直接给最终答案。"
            "请只输出 JSON，不要输出其他文本。\n"
            "输出字段要求：\n"
            "{\n"
            '  "thought_summary": "一句话说明当前判断（可展示）",\n'
            '  "decision": "tool" 或 "final",\n'
            '  "tool_name": "当 decision=tool 时填写",\n'
            '  "tool_args": { },\n'
            '  "final_answer": "当 decision=final 时填写"\n'
            "}\n"
            "规则：\n"
            "- 当 decision=tool 时，只能从以下工具中选择：get_current_time, serper_search\n"
            "- 当 decision=final 时，给 final_answer，且 tool_name 置空\n"
            "- 不要输出内部私密推理，只给可展示摘要"
        )

        planning_messages = [
            {"role": "system", "content": planner_prompt},
            *state_messages,
        ]

        plan_resp = client.chat.completions.create(
            model=model,
            messages=planning_messages,  # type: ignore[arg-type]
            temperature=0.2,
        )
        plan_text = (plan_resp.choices[0].message.content or "").strip()
        plan = parse_json_safely(plan_text)
        if not plan:
            return f"规划输出不是 JSON，停止。原始输出：{plan_text}"

        thought_summary = str(plan.get("thought_summary", "")).strip()
        decision = str(plan.get("decision", "")).strip()
        print(f"\n=== Step {step} ===")
        print(f"[思考摘要] {thought_summary or '(空)'}")
        print(f"[规划JSON] {json.dumps(plan, ensure_ascii=False)}")

        if decision == "final":
            final_answer = str(plan.get("final_answer", "")).strip()
            print("\n=== Final Answer ===")
            print(final_answer)
            return final_answer

        if decision != "tool":
            return f"无效 decision: {decision}，停止。"

        tool_name = str(plan.get("tool_name", "")).strip()
        tool_args = plan.get("tool_args", {})
        if not isinstance(tool_args, dict):
            return "tool_args 不是对象，停止。"
        if tool_name not in allowed_tool_names:
            return f"工具名不在白名单: {tool_name}，停止。"

        print(f"[行动] 调用工具: {tool_name} args={tool_args}")

        # 执行工具
        try:
            tool_result = call_tool(tool_name, tool_args)
        except Exception as e:
            tool_result = f"工具异常: {e}"

        observation = summarize_observation(str(tool_result))
        print(f"[观察摘要] {observation}")

        # 把本轮可用事实回填给后续 step
        state_messages.append(
            {
                "role": "assistant",
                "content": (
                    f"step={step}; thought_summary={thought_summary}; "
                    f"action={tool_name}({json.dumps(tool_args, ensure_ascii=False)}); "
                    f"observation={observation}"
                ),
            }
        )

    return "达到最大步数仍未完成，停止。"


def main() -> None:
    q = input("请输入问题：").strip() or "现在几点？并搜索一下南京今天天气。"
    print("\n--- ReAct 可展示过程开始 ---")
    run_react_with_visible_trace(q)

if __name__ == "__main__":
    main()
```


## 怎么写好一个 ReAct 过程

- 明确状态：goal、known_facts、missing_info、step_count
- 明确每步输出结构：thought_summary、decision、action、action_input
- 强制单步单动作：每轮只允许一个工具调用或 final
- 工具白名单 + 参数校验：防止幻觉工具和坏参数
- Observation 压缩：长结果先摘要，再保留原文可追溯
- 停止条件：max_steps、重复检测、置信度阈值
- 错误策略：超时/失败重试上限 + 降级回答
- 可观测性：每步日志（计划、工具输入、结果、耗时）
