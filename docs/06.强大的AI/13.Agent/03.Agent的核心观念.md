## Agent的实际用处

- 给模型提供工具能力（搜索、代码执行、文件系统等）
- 让模型在多步循环里“观察 -> 决策 -> 调用工具 -> 再观察”
- 管理状态（消息、任务、记忆、中断恢复）
- 提供工程能力（日志、权限、并发、容错）


## 没有框架如何构建一个Agent

只用 一个OpenAI 接口，怎么做一个 Agent。最小闭环是：

1. 收到用户问题
2. 把当前上下文 + 工具描述发给模型
3. 模型要么直接回答，要么返回“我要调用某个工具”
4. 你的程序代码执行工具
5. 把工具结果回填给模型
6. 重复 2~5，直到模型给最终回答（或达到步数上限）

这就是最核心的 Agent Loop。

## 最小循环示例代码

```python
from __future__ import annotations

import json
from datetime import datetime
from typing import Any

from openai import OpenAI

def get_current_time() -> str:
    """工具1：返回当前本地时间。"""
    return datetime.now().strftime("%Y-%m-%d %H:%M:%S")

def serper_search(query: str) -> str:
    """
    工具2：搜索工具。
    """
    return f"[mock_serper_result] query={query}"


# 提供给模型的“工具声明”
TOOLS: list[dict[str, Any]] = [
    {
        "type": "function",
        "function": {
            "name": "get_current_time",
            "description": "获取当前时间",
            "parameters": {"type": "object", "properties": {}, "required": []},
        },
    },
    {
        "type": "function",
        "function": {
            "name": "serper_search",
            "description": "互联网搜索，输入 query，输出检索结果摘要",
            "parameters": {
                "type": "object",
                "properties": {
                    "query": {"type": "string", "description": "搜索关键词"},
                },
                "required": ["query"],
            },
        },
    },
]


def run_agent(user_input: str, model: str = "qwen-max", max_steps: int = 8) -> str:

    client = OpenAI(
        api_key="sk-xxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxxx",
        base_url="https://dashscope.aliyuncs.com/compatible-mode/v1",
    )

    # 会话状态：每一步都追加到 messages，模型才能临时“记住”上下文
    messages: list[dict[str, Any]] = [
        {"role": "system", "content": "你是一个会使用工具的助手，先查证再回答。"},
        {"role": "user", "content": user_input},
    ]

    for step in range(1, max_steps + 1):
        print(f"\n=== Step {step} ===")

        # 让模型自己决定是否调用工具
        resp = client.chat.completions.create(
            model=model,
            messages=messages,
            tools=TOOLS,
            tool_choice="auto",
            temperature=0.2,
        )
        msg = resp.choices[0].message

        # 分支A：模型要调用工具
        if msg.tool_calls:
            # 先把 assistant 的 tool_calls 写入上下文
            messages.append(
                {
                    "role": "assistant",
                    "content": msg.content or "",
                    "tool_calls": [
                        {
                            "id": tc.id,
                            "type": "function",
                            "function": {
                                "name": tc.function.name,
                                "arguments": tc.function.arguments,
                            },
                        }
                        for tc in msg.tool_calls
                    ],
                }
            )

            # 逐个执行工具，并把结果以 tool 消息写回
            for tc in msg.tool_calls:
                tool_name = tc.function.name
                args = json.loads(tc.function.arguments or "{}")
                print(f"[工具调用] {tool_name} args={args}")

                try:
                    if tool_name == "get_current_time":
                        tool_result = get_current_time()
                    elif tool_name == "serper_search":
                        tool_result = serper_search(args["query"])
                    else:
                        tool_result = f"工具不存在: {tool_name}"
                except Exception as e:
                    tool_result = f"工具执行异常: {e}"

                print(f"[工具结果] {str(tool_result)[:200]}")

                messages.append(
                    {
                        "role": "tool",
                        "tool_call_id": tc.id,
                        "content": str(tool_result),
                    }
                )
            # 回到下一轮，让模型基于工具结果继续推理
            continue

        # 分支B：模型给出最终答案
        final_answer = msg.content or ""
        print("\n=== Final Answer ===")
        print(final_answer)
        return final_answer

    return "达到最大推理步数，已停止。"


def main() -> None:
    question = input("请输入问题：").strip() or "现在几点？顺便搜索一下南京天气。"
    run_agent(question)


if __name__ == "__main__":
    main()
```

如果你要说你会写一个 Agent，那么这个步骤的逻辑就足够了。 但你要说你能把Agent用到实际的项目里，那么你需要考虑更多的问题：

上面这段代码虽然已经具备“Agent”核心，但还缺工程化能力：
- 有了：工具调用、多步循环、上下文回填
- 还缺：记忆持久化、权限隔离、日志追踪、HITL 审批、并发子任务、故障恢复等

常见的 Agent 框架的价值就是把这些“还缺”的部分标准化，框架的价值不在“能不能做”，而在“能不能稳定地长期做”。
